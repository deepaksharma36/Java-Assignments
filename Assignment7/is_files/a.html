            package checkpath;

            import java.util.LinkedList;

            import java.util.concurrent.atomic.AtomicBoolean;

            import java.util.Queue;

            import java.util.Stack;

            /**

            *

            * @author hpatel

            *

            */

            public class Graph {

            /* Hard code total 5 vertices in this graph */

            int n;

            /* Create adjacency matrix staright away when any object with any constructor */

            int AdjMat[][];

            /* State of vertice which checks it got visited or not */

            boolean visited[];

            /**

            * Initialize graph when Graph gets created

            */

            public Graph(int n) {

            this.n = n;

            AdjMat = new int[n][n];

            visited = new boolean[n];

            init();

            }

            /**

            * Initialize adjacency matrix which has connectivity information and visited state

            * to false.

            */

            private void init() {

            for (int i = 0; i < n; i++) {

            for (int j = 0; j < n; j++) {

            AdjMat[i][j] = 0;

            }

            visited[i] = false;

            }

            }

            public void makeGraph() {

            addEdge(0, 1);

            addEdge(0, 2);

            addEdge(1, 2);

            addEdge(2, 0);

            addEdge(2, 3);

            addEdge(3, 3);

            }

            /**

            * Add edge between given two vertices.

            *

            * @param i

            * @param j

            */

            public void addEdge(int i, int j) {

            /* Make sure i and j are not outof bound */

            assert i < n && j < n;

            /* The graph is directed */

            AdjMat[i][j] = 1;

            }

            /**

            * Print graph.

            */

            public void printGraph() {

            for (int i = 0; i < n; i++) {

            for (int j = 0; j < n; j++) {

            if (AdjMat[i][j] > 0) {

            System.out.println(i + "-->" + j);

            }

            }

            }

            }

            /**

            * Method to check whether there is a path between two

            * vertices.

            */

            public boolean checkPath(int v1, int v2) {

            /* Vertices has to in range */

            assert v1 < n && v2 < n;

            /**

            * If both vertices is same then if self loop exist then

            * there there a path to itself otherwise not.

            */

            if (v1 == v2) {

            if (checkSelfLoop(v1, v2)) {

            return true;

            } else {

            return false;

            }

            }

            /**

            * Traverse twice and if both flags are true then there is a path

            * between these two starting from either two.

            */

            return dfs(v1, v2) & dfs(v2, v1);

            }

            /**

            * Stack alternative for recursive version.

            *

            * @param i

            */

            private boolean dfs(int v1, int v2) {

            Stack<integer> s = new Stack<integer>();

            s.push(v1);

            visited[v1] = true;

            System.out.print("Visting : " + v1);

            while (!s.isEmpty()) {

            /* get top stack element */

            int n = s.peek().intValue();

            /* get first unvisited child */

            int c = getUnvisitedChild(n);

            if (c != -1) {

            /* there is a path to V2 hence assign flag true and then clear

            * visited state of all vertices of graph and stop process of

            * checking path to v2 as we found path.

            */

            System.out.print("-->" + c);

            if (c == v2) {

            clearVisited();

            return true;

            }

            s.push(c);

            visited[c] = true;

            } else {

            /**

            * There is no unvisited child left for this node. Hence pop

            * element from stack.

            */

            s.pop();

            }

            }

            clearVisited();

            return false;

            }

            /**

            * Method to check if self loop exist for given vertice

            * @param v1

            * @param v2

            * @return

            */

            private boolean checkSelfLoop(int v1, int v2) {

            if (AdjMat[v1][v2] > 0 && AdjMat[v2][v1] > 0)

            return true;

            return false;

            }

            /**

            * Get First unvisited child for this vertice. This is used by DFS

            * @param v

            * @return

            */

            private int getUnvisitedChild(int v) {

            int j = -1;

            for (int i = 0; i < n; i++) {

            if (AdjMat[v][i] > 0 && !visited[i]) {

            return i;

            }

            }

            return j;

            }

            /**

            * Method to use breadth first traversal. It explores all children of node and then

            * only children of children. It goes in level by level.

            * @param i

            */

            public void bfs(int i) {

            Queue<integer> q = new LinkedList<integer>();

            q.add(i);

            while (!q.isEmpty()) {

            visitNode(q);

            }

            clearVisited();

            }

            private void visitNode(Queue<integer> q) {

            Integer v = (Integer) q.remove();

            if (!visited[v]) {

            visited[v.intValue()] = true;

            System.out.println("Visited : " + v.intValue());

            }

            addAllChild(q, v.intValue());

            }

            private void addAllChild(Queue<integer> q, int v) {

            for (int i = 0; i < n; i++) {

            if (AdjMat[v][i] > 0 && !visited[i]) {

            q.add(i);

            }

            }

            }

            /**

            * Method to clear visited state of vertice after traversal.

            */

            private void clearVisited() {

            for (int i = 0; i < n; i++) {

            visited[i] = false;

            }

            }

            }

            /**

            * Traversal : Is a problem in computing which relates to visitation of nodes in any data structure.

            * Graph : Directional graph : It has no direction constraint. If there is an edge(i,j) then edge(j,i)

            * also true.

            * Adjacency matrix : 2D array. If N is number of vertices then matrix of size AdjMat[N][N]

            *

            * DFS : Depth First Search : It go to depth and then comes backward and visit nodes. (Stack)

            * BFS : Bredth First Search : It first goes to immediate children and then to further children. (Queue)

            *

            *

            **/
